- [ ] **Step 1: Create `update-data.py` with PoE 1 gem fetching**

```python
#!/usr/bin/env python3
"""Fetches PoE game data from community sources and writes JSON files."""

import json
import os
import urllib.request

DATA_DIR = os.path.join(os.path.dirname(os.path.abspath(__file__)), "data")

REPOE_GEMS_URL = "https://raw.githubusercontent.com/brather1ng/RePoE/master/RePoE/data/gems.json"
REPOE_GEM_TAGS_URL = "https://raw.githubusercontent.com/brather1ng/RePoE/master/RePoE/data/gem_tags.json"


def fetch_json(url):
    req = urllib.request.Request(url, headers={"User-Agent": "RandomOfExile/1.0"})
    with urllib.request.urlopen(req) as resp:
        return json.loads(resp.read().decode())


def build_poe1_gems():
    print("Fetching PoE 1 gem data from RePoE...")
    raw_gems = fetch_json(REPOE_GEMS_URL)
    tag_names = fetch_json(REPOE_GEM_TAGS_URL)

    gems = []
    for key, gem in raw_gems.items():
        if gem.get("is_support", True):
            continue
        if gem.get("base_item", {}).get("release_state") != "released":
            continue

        display_name = gem.get("active_skill", {}).get("display_name")
        if not display_name:
            display_name = gem.get("base_item", {}).get("display_name", key)

        raw_tags = gem.get("tags", [])
        display_tags = []
        for t in raw_tags:
            mapped = tag_names.get(t)
            if mapped and mapped.get("name"):
                display_tags.append(mapped["name"])

        gems.append({"name": display_name, "tags": display_tags})

    # Deduplicate by name (some gems have multiple internal entries)
    seen = set()
    unique_gems = []
    for g in gems:
        if g["name"] not in seen:
            seen.add(g["name"])
            unique_gems.append(g)

    unique_gems.sort(key=lambda g: g["name"])
    return unique_gems


if __name__ == "__main__":
    os.makedirs(DATA_DIR, exist_ok=True)

    poe1_gems = build_poe1_gems()
    with open(os.path.join(DATA_DIR, "poe1-gems.json"), "w") as f:
        json.dump(poe1_gems, f, indent=2)
    print(f"  PoE 1 gems: {len(poe1_gems)}")
```

- [ ] **Step 1: Add the slot machine spin logic**

Append this after the `loadData(state.game);` line at the end of the script:

```javascript
const ITEM_HEIGHT = 60;
const SPIN_DURATION = 2500; // ms
const SPIN_STAGGER = 500; // ms delay between gem and ascendancy stopping

function shuffle(arr) {
    const copy = [...arr];
    for (let i = copy.length - 1; i > 0; i--) {
        const j = Math.floor(Math.random() * (i + 1));
        [copy[i], copy[j]] = [copy[j], copy[i]];
    }
    return copy;
}

function buildStrip(stripEl, items, winnerIndex) {
    stripEl.innerHTML = "";
    items.forEach((item) => {
        const div = document.createElement("div");
        div.className = "slot-item";
        div.textContent = typeof item === "string" ? item : item.name;
        stripEl.appendChild(div);
    });
    // Position so the first item is visible
    stripEl.style.transition = "none";
    stripEl.style.transform = `translateY(0px)`;
}

function animateSpin(stripEl, totalItems, winnerIndex, duration) {
    return new Promise((resolve) => {
        const totalDistance = (totalItems - 1 - winnerIndex) * ITEM_HEIGHT;
        const startTime = performance.now();

        // Start from the top
        stripEl.style.transition = "none";
        stripEl.style.transform = `translateY(0px)`;

        function frame(now) {
            const elapsed = now - startTime;
            const progress = Math.min(elapsed / duration, 1);
            // Ease out cubic for deceleration
            const eased = 1 - Math.pow(1 - progress, 3);
            const y = -eased * totalDistance;
            stripEl.style.transform = `translateY(${y}px)`;

            if (progress < 1) {
                requestAnimationFrame(frame);
            } else {
                resolve();
            }
        }

        requestAnimationFrame(frame);
    });
}

async function spin() {
    if (state.spinning || state.gems.length === 0 || state.ascendancies.length === 0) return;
    state.spinning = true;
    elements.randomizeBtn.disabled = true;
    elements.togglePoe1.disabled = true;
    elements.togglePoe2.disabled = true;
    elements.results.classList.remove("visible");

    // Pick winners
    const gemWinner = state.gems[Math.floor(Math.random() * state.gems.length)];
    const ascWinner = state.ascendancies[Math.floor(Math.random() * state.ascendancies.length)];

    // Build strips: shuffle a bunch of items with the winner placed at a specific position near the end
    const gemPool = shuffle(state.gems);
    const ascPool = shuffle(state.ascendancies);

    // Repeat items to create a long scrolling strip, place winner near end
    const repeatCount = 6;
    const gemItems = [];
    const ascItems = [];
    for (let i = 0; i < repeatCount; i++) {
        gemItems.push(...shuffle(state.gems));
        ascItems.push(...shuffle(state.ascendancies));
    }
    // Place winner as the last item
    gemItems.push(gemWinner);
    ascItems.push(ascWinner);
    const gemWinnerIndex = gemItems.length - 1;
    const ascWinnerIndex = ascItems.length - 1;

    buildStrip(elements.gemStrip, gemItems, gemWinnerIndex);
    buildStrip(elements.ascStrip, ascItems, ascWinnerIndex);

    // Animate both, gem stops first, ascendancy stops after stagger
    await Promise.all([
        animateSpin(elements.gemStrip, gemItems.length, gemWinnerIndex, SPIN_DURATION),
        animateSpin(elements.ascStrip, ascItems.length, ascWinnerIndex, SPIN_DURATION + SPIN_STAGGER),
    ]);

    showResults(gemWinner, ascWinner);

    state.spinning = false;
    elements.randomizeBtn.disabled = false;
    elements.togglePoe1.disabled = false;
    elements.togglePoe2.disabled = false;
}

elements.randomizeBtn.addEventListener("click", spin);

function wikiUrl(name) {
    const base =
        state.game === "poe1"
            ? "https://www.poewiki.net/wiki/"
            : "https://www.poe2wiki.net/wiki/";
    return base + encodeURIComponent(name.replace(/ /g, "_"));
}

function ninjaUrl(gemName, ascName) {
    const base =
        state.game === "poe1"
            ? "https://poe.ninja/challenge/builds"
            : "https://poe.ninja/poe2/builds";
    return `${base}?skill=${encodeURIComponent(gemName)}&class=${encodeURIComponent(ascName)}`;
}

function showResults(gem, ascendancy) {
    elements.resultGemName.textContent = gem.name;
    elements.resultGemTags.textContent = gem.tags.join(", ");
    elements.resultAscName.textContent = ascendancy.name;
    elements.resultAscClass.textContent = ascendancy.class;

    elements.resultLinks.innerHTML = "";

    const links = [
        { text: `${gem.name} Wiki`, href: wikiUrl(gem.name) },
        { text: `${ascendancy.name} Wiki`, href: wikiUrl(ascendancy.name) },
        {
            text: "Builds on poe.ninja",
            href: ninjaUrl(gem.name, ascendancy.name),
        },
    ];

    links.forEach(({ text, href }) => {
        const a = document.createElement("a");
        a.href = href;
        a.textContent = text;
        a.target = "_blank";
        a.rel = "noopener noreferrer";
        elements.resultLinks.appendChild(a);
    });

    elements.results.classList.add("visible");
}
```
